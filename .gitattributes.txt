# Set default behavior to automatically normalize line endings
* text=auto

# Shell scripts should always use LF
*.sh text eol=lf

# Explicitly declare files that should always have LF
*.yml text eol=lf
*.yaml text eol=lf
Dockerfile text eol=lf

# Declare files that should always have CRLF
*.bat text eol=crlf

# Denote all files that are truly binary
*.png binary
*.jpg binary
